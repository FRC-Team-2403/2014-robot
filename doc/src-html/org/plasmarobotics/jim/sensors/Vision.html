013/**
014 *This class handles the receiving of visual information
015 * @author jim
016 */
017public class Vision {
018    private NetworkTable server;
019    private double distance; //distance from the target
020    private int pixelsOff; //negative for left positive for right
021    
022    public Vision(){
023        server = NetworkTable.getTable("SmartDashboard");
024        System.out.println("The Eye is open");
025        
026        
027    }
028    
029    /**
030     * called constantly during autonomous
031     */
032    public void update(){
033        SmartDashboard.putNumber("Hot: ", server.getNumber("Hot", 0.0));
034        
035    }
036    
037    /**
038     * @return whether the goal is hot or not
039     */
040    public boolean isHot(){
041        double serverInput = server.getNumber("Hot", 0.0);
042        if(serverInput == 1)
043            return true;
044        else 
045            return false;
046    }
047}
