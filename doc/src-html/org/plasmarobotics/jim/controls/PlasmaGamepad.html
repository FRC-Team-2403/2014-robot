001package org.plasmarobotics.jim.controls;
002
003import edu.wpi.first.wpilibj.Joystick;
004
005/**
006 *
007 * @author Allek
008 */
009public class PlasmaGamepad extends Joystick {
010    
011     private final ToggleableButton aButton = new ToggleableButton(this, 1),
012            bButton = new ToggleableButton(this, 2),
013            xButton = new ToggleableButton(this, 3),
014            yButton = new ToggleableButton(this, 4),
015            leftBumper = new ToggleableButton(this, 5),
016            rightBumper = new ToggleableButton(this, 6),
017            backButton = new ToggleableButton(this, 7),
018            startButton = new ToggleableButton(this, 8),
019            leftJoystickButton = new ToggleableButton(this, 9),
020            rightJoystickButton = new ToggleableButton(this, 10);
021            
022            final TriggerButton leftTriggerButton = new TriggerButton(this, 3, false),
023                    rightTriggerButton = new TriggerButton(this, 3, true),
024                    rightDPadButton = new TriggerButton(this, 6, true),
025                    leftDPadButton = new TriggerButton(this, 6, false);
026    
027    public PlasmaGamepad(int port) {
028        super(port);
029    }
030    public ToggleableButton getAButton() {
031        return this.aButton;
032    }
033    public ToggleableButton getBButton() {
034        return this.bButton;
035    }
036    public ToggleableButton getXButton() {
037        return this.xButton;
038    }
039    public ToggleableButton getYButton() {
040        return this.yButton;
041    }
042    public ToggleableButton getLeftBumper() {
043        return this.leftBumper;
044    }
045    public ToggleableButton getRightBumper() {
046        return this.rightBumper;
047    }
048    public ToggleableButton getBackButton() {
049        return this.backButton;
050    }
051    public ToggleableButton getStartButton() {
052        return this.startButton;
053    }
054    /**
055     * Get left joystick button (pressed in or not).
056     **/
057    public ToggleableButton getLeftJoystickButton() {
058        return this.leftJoystickButton;
059    }
060    /**
061     * Get right joystick button (pressed in or not).
062     **/
063    public ToggleableButton getRightJoystickButton() {
064        return this.rightJoystickButton;
065    }
066    
067    /**
068     * Get left joystick x-axis (left-right).
069     * Full left = -1, full right = +1.
070     **/
071    public double getLeftJoystickXAxis() {
072        return super.getRawAxis(1);
073    }
074    /**
075     * Get left joystick y-axis (up-down).
076     * Full up = -1, full down = +1.
077     **/
078    public double getLeftJoystickYAxis() {
079        return super.getRawAxis(2);
080    }
081    /**
082     * Get right joystick x-axis (left-right).
083     * Full left = -1, full right = +1.
084     **/
085    public double getRightJoystickXAxis() {
086        return super.getRawAxis(4);
087    }
088    /**
089     * Get right joystick y-axis (up-down).
090     * Full up = -1, full down = +1.
091     **/
092    public double getRightJoystickYAxis() {
093        return super.getRawAxis(5);
094    }
095    /**
096     * Get trigger axis.
097     * Right depressed -> -1
098     * Left depressed -> +1
099     * Both fully depressed/neither touched = 0
100     * @return Trigger axis value
101     **/
102    public double getTriggerAxis() {
103        
104        return super.getRawAxis(3);
105    }
106    /**
107     * Check if right trigger is pressed (trigger axis < 0).
108     * @deprecated Prone to looping issues. Use
109     * PlasmaGamepad.getRightTriggerButton().get() instead.
110     * @return True if pressed
111     **/
112    public boolean rightTriggerPressed() {
113        return (getTriggerAxis() < 0);
114    }
115    /**
116     * Check if left trigger is pressed (trigger axis > 0).
117     * @deprecated Prone to looping issues. Use
118     * PlasmaGamepad.getLeftTriggerButton().get() instead.
119     * @return True if pressed
120     **/
121    public boolean leftTriggerPressed() {
122        return (getTriggerAxis() > 0);
123    }
124    /**
125     * Get DPad axis.
126     * Right = 1
127     * Left = -1
128     * @return DPad axis
129     **/
130    public int getDPadXAxis() {
131        return (int) (super.getRawAxis(6));
132    }
133    /**
134     * Check if D-Pad is pressed right.
135     * @deprecated Prone to looping issues. Use
136     * PlasmaGamepad.getRightDPadButton().get() instead.
137     * @return True if pressed
138     **/
139    public boolean rightDPadPressed() {
140        return (getDPadXAxis() == 1);
141    }
142    /**
143     * Check if D-Pad is pressed left.
144     * @deprecated Prone to looping issues. Use
145     * PlasmaGamepad.getLeftDPadButton().get() instead.
146     * @return True if pressed
147     **/
148    public boolean leftDPadPressed() {
149        return (getDPadXAxis() == -1);
150    }
151    public TriggerButton getRightDPadButton() {
152        return this.rightDPadButton;
153    }
154    public TriggerButton getLeftDPadButton() {
155        return this.leftDPadButton;
156    }
157    public TriggerButton getRightTriggerButton() {
158        return this.rightTriggerButton;
159    }
160    public TriggerButton getLeftTriggerButton() {
161        return this.leftTriggerButton;
162    }
163}
